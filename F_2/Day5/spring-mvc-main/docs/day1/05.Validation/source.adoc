package com.nhnacademy.springmvc.domain;

import lombok.Value;
import org.hibernate.validator.constraints.Length;

import javax.validation.constraints.NotBlank;

@Value
public class UserRegisterRequest {
    @NotBlank
    String id;
    @Length(min =8, max = 20)
    String password;
    @Length(min = 0)
    int age;

    // TODO #3: 이름 필드 추가
    @NotBlank
    @Length(min = 1)
    String name;

}
